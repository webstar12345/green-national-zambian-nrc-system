# Git Deployment Commands
# Copy and paste these commands one by one into Git Bash

# Step 1: Check if git is working
git --version

# Step 2: Configure git (replace with your info)
git config --global user.name "Your Name"
git config --global user.email "your-email@example.com"

# Step 3: Initialize git repository
git init

# Step 4: Add all files
git add .

# Step 5: Commit
git commit -m "Initial commit - Zambian NRC System ready for deployment"

# Step 6: Add GitHub remote (REPLACE YOUR_USERNAME with your actual GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/zambian-nrc-system.git

# Step 7: Rename branch to main
git branch -M main

# Step 8: Push to GitHub
git push -u origin main

# If you get authentication error, you'll need to create a Personal Access Token:
# 1. Go to: https://github.com/settings/tokens
# 2. Click "Generate new token (classic)"
# 3. Select "repo" scope
# 4. Copy the token and use it as password when prompted
